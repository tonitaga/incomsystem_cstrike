        // Воспроизведение звука при загрузке страницы
        window.addEventListener('load', function() {
            const audio = document.getElementById('greetingSound');
            
            // Попытка воспроизвести звук
            const playPromise = audio.play();
            
            if (playPromise !== undefined) {
                playPromise.then(_ => {
                    // Автовоспроизведение успешно
                    console.log("Звук приветствия воспроизводится");
                }).catch(error => {
                    // Автовоспроизведение не удалось
                    console.log("Автовоспроизведение заблокировано. Звук будет воспроизведен после взаимодействия с пользователем.");
                    
                    // Добавляем обработчик клика для воспроизведения звука
                    document.body.addEventListener('click', function playOnClick() {
                        audio.play();
                        document.body.removeEventListener('click', playOnClick);
                    });
                });
            }
        });
    </script>
